\section{バグ検出を考える上でのJulia言語の考察}

コード example

\begin{minted}[frame=lines,linenos,breaklines]{julia}
  abstract type T end
  struct T1 <: T
      foo::Int
      bar
  end
  struct T2 <: T
      foo::Float64
      baz
  end
  function foo(t::T)
      if t.foo <= 0
          return nothing
      else
          return t.bar
      end
  end

  foo(T1(1, 1))
  foo(T2(1., 1))
  foo(T2(0., 1))
\end{minted}
